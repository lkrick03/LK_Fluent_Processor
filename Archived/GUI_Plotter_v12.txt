            ## Time Averaging function that is deployed if "Cumulative TA" is selected

            def TA(flow_time, dataset, std):

                temp = []
                sum_x = dataset[0]
                sum_x2 = dataset[0]**2
                stdev = []
                ta = dataset[0]
                n = 1
                
                for j in range(1,len(dataset)):
                    
                    tp_a = flow_time[j] - flow_time[0]
                    dt = flow_time[j] - flow_time[j-1]
                    tp = tp_a
                    ti = tp+dt
                    ta = (ta*tp + dt*dataset[j]) / ti
                    temp.append(ta)

                    if std == True:
                        
                        n += 1
                        sum_x += dataset[j]
                        sum_x2 += dataset[j]**2
                        mean = sum_x / n
                        stdev.append(((sum_x2 / n) - (mean * mean))**0.5)
                
                return temp, stdev

        def fft_analysis_pad_no_plot(y, N, T): 
            y_clean = np.zeros(len(y))
            y_clean[:] = y[:] - np.mean(y[:]) # Remove the mean (eliminates point) 
            NFFT = 2**(math.ceil(math.log(len(y_clean), 2))) # Calculate number of points with the next highest power of 2 
            yf = fft(y_clean, NFFT) # Compute the complex FFT including zero padding up to next power of 2
            xf = fftfreq(NFFT, T)[:NFFT//2] # Generate frequency data
            yf_amp = (2.0/N)*np.abs(yf[:NFFT//2]) # Compute the single-sided amplitude spectrum of y 

            yf_max_loc = np.argmax(yf_amp) # Find location of max amplitude
            xf_peak = xf[yf_max_loc] # Find corresponding frequency value 
            yf_max = max(yf_amp) 

            return xf, yf_amp, xf_peak, yf_max
