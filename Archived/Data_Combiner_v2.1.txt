###### Data Combiner ######
# v1:   Base data combiner.     5/2/2023
# v2:   Added functionality to detect if there is an inconsistency in the number of columns for combined datasets.     9/25/2024
#       If the columns are not equal, columns of zeros are added to all datasets with a number of columns less than 
#       the maximum until all datasets are able to be combined.
# v2.1: Uneven datasets are now appended with blank columns instead of columns of zeros.   9/25/2024

from tkinter import filedialog
import numpy as np
from tkinter import simpledialog

num = simpledialog.askinteger(title = 'How Many Files to Combine?', prompt = 'How Many Files to Combine?')

files = []

for i in range(num):

    temp = filedialog.askopenfilename(title="Select Data File to Be Read")
    files.append(temp)

def trim_overlap_data(data, next_data):
    loc = np.argwhere(data[:,0] >= next_data[0,0]) # Find the indices for positions of overlap 
    if loc.size > 0: # If the is indeed overlap 
        data = np.delete(data, slice(int(loc[0]),len(data)), axis=0) # Remove all overlapping rows 
    return data 

data_arrays = []

for i in range(num):

    f = open(files[i])
    data_arrays.append(np.genfromtxt(f, delimiter=' ', skip_header=3))
    f.close()

for i in range(0,num-1):
    data_arrays[i] = trim_overlap_data(data_arrays[i], data_arrays[i+1]) 

size = []
for i in range(len(data_arrays)):
    size.append(data_arrays[i].shape[1])

columns = max(size)

for i in range(len(data_arrays)):
    if data_arrays[i].shape[1] != columns:
        diff = columns - data_arrays[i].shape[1]
        temp = data_arrays[i]
        zero = np.full((data_arrays[i].shape[0], diff), ' ')
        data_arrays[i] = np.concatenate((temp, zero), 1)

data = data_arrays[0] 

for i in range(1,num):
    data = np.append(data, data_arrays[i], axis=0) 

data = np.ndarray.tolist(data)

W = open('Final_Data.txt', 'w+')

for i in range(len(data)):
    for j in range(len(data[i])):
        W.write(str(data[i][j]) + ' ')
    W.write('\n')

W.close()
